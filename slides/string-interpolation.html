String Interpolation


function usingStringInterpolation(){
  const user = {
    firstName: "Matt",
    lastName: "Thomas",
    city: 'Milwaukee'
    state: 'WI'
    country: "USA"
  }

  const singleLine = `{user.firstName} {user.lastName} from {user.country}`

  const multiLine = `{user.firstName} {user.lastName}
    {user.city} {user.state}
    {user.country}`
}
