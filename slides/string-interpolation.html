`String Interpolation and Multi-Line`


let user = {
  name: "John Smith",
  address: "5739 Laurel Lane",
  city: "Lawndale",
  state: "CA",
  postalCode: "90260"
}

const address = `${user.name}
${user.address}
${user.city} ${user.state} ${user.postalCode}`

console.log(address)
/*
John Smith
5739 Laurel Lane
Lawndale CA 90260
*/
