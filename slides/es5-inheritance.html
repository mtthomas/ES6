ES5 Inheritance


function Dog(){
  Mammal.call(this, 4)
}

Dog.prototype = new Mammal()
Dog.prototype.constructor = Dog

Dog.prototype.bark = function(){
  return "bark"
}

const dog = new Dog()
console.log(dog.numberOfLegs) //4
console.log(dog.walk()) //walking
console.log(dog.bark()) //bark
console.log(dog instanceof Dog) //true
  • Working with functions and not classes, Mammal.call
  • Need to remember to set the prototype and constructor
  • One or many ways of defining inheritence
