Object Literals


function generateId(){
  console.log("generating an id")
}

const prefix = "prefix"

let factory = {
  generateId, //generateId: generateId
  [prefix + "_message"]: "message", //dynamic properties
  suffix: "suffix",
  toString(){ //shorthand for creating functions
    return "factory"
  }
}

console.log(factory.generateId()) //generating an id
console.log(factory.suffix) //suffix
console.log(factory.toString()) //factory
console.log(factory.prefix_message) //message
