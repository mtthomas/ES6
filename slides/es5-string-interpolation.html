ES5 String Interpolation and Multi-Line


let user = {
  name: "John Smith",
  address: "5739 Laurel Lane",
  city: "Lawndale",
  state: "CA"
}

const postalCode: "90260"

const address = "SHIP TO:" + user.name + "\n"
  + user.address + "\n"
  + user.city + " " + user.state + " " + postalCode

console.log(address)
/*
John Smith
5739 Laurel Lane
Lawndale CA 90260
*/
