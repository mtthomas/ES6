function Person(){
  var self = this;

  this.name = "John Smith";

  setInterval(function(){
    //this is undefined
    console.log(self.name)
  }, 1000)
}

function Person(){
  this.name = "John Smith";

  setInterval(() => {
    console.log(this.name)
  }, 1000)
}

Arrow functions capture the this context of their parent
