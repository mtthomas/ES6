Destructuring


function createUser(){
  return {
    firstName: "John",
    lastName: "Smith",
    userId: "JS2016",
    startDate: "1/1/2016"
  }
}

let {
  firstName: firstName,
  lastName,
  userId: uid
} = createUser()

console.log(firstName) //John
console.log(lastName) //Smith
console.log(uid) //JS2016

  • Easier than prefixings each variable
