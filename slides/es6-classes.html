ES6 Classes


class Mammal{
  constructor(numberOfLegs){
    this._numberOfLegs = numberOfLegs
  }
  get numberOfLegs(){
    return this._numberOfLegs
  }
  set numberOfLegs(){
    this._numberOfLegs = numberOfLegs
  }
  walk(){ return "walking" }
}

const mammal = new Mammal(2)
console.log(mammal.numberOfLegs) //2
console.log(mammal.walk()) //walking
