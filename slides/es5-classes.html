ES5 Classes


function Mammal(numberOfLegs){
  this.numberOfLegs = numberOfLegs;
}

Mammal.prototype.walk = function() { return "walking" }

const mammal = new Mammal(2)
console.log(mammal.numberOfLegs) //2
console.log(mammal.walk()) //walking
