Prototypical Inheritance

  • Prototype chain
  • Lack of a super keyword
  • Multiple ways to inherit

function Mammal(){
  /* initialize Mammal */
}
Mammal.prototype.speak = function(){
  /* speak */
}

function Dog(){
  /* initialize Dog */
}
Dog.prototype = new Mammal();
Dog.prototype.constructor = Dog;

var dog = new Dog();
dog.speak();
